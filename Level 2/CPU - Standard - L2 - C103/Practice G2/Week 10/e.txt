#include <bits/stdc++.h>
using namespace std;
using ll = long long;

int main() {
  ios_base::sync_with_stdio(0);cin.tie(0);cout.tie(0);
  int t;cin>>t;
  while(t--) {
    ll n;cin>>n;
    ll mx = -1;
    for(ll i = 2;i * i <= n;i++) {
      if(n % i == 0) {
        mx = max(mx , i);
      }
      while(n % i == 0)n /= i;
    }
    if(n != 1)mx = max(mx , n);
    cout<<mx<<'\n';
  }
}
